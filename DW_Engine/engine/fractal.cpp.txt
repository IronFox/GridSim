EveFractalMapView::EveFractalMapView():velocity(0),lod(1.0),visual_range(100),most_detail(UNSIGNED_UNDEF),visible_sectors(0),detail_limit(0)
{
    __eye4(system);
    _c16(system,invert);
    _v3(movement_direction,1,0,0);
    _clear(texture_root);
    _v3(texture_x,1,0,0);
    _v3(texture_y,0,1,0);
    _v3(texture_up,0,0,1);
}

void EveFractalMapView::begin()
{
    BEGIN
    visible_sectors = 0;
    most_detail = UNSIGNED_UNDEF;
    END
}


void EveFractalMapView::setSystem(const float sys[16])
{
    BEGIN
    _c16(sys,system);
    __invertSystem(system,invert);
    END
}

void EveFractalMapView::updateInvert()
{
    __invertSystem(system,invert);
}


const float* EveFractalMapView::origin() const
{
    return system+12;
}

void EveFractalMapView::resolveESX(float target[3]) const
{
    float temp[3];
    __rotate(system,texture_x,temp);
    __rotate(camera.modelview,temp,target);
}

void EveFractalMapView::resolveESY(float target[3]) const
{
    float temp[3];
    __rotate(system,texture_y,temp);
    __rotate(camera.modelview,temp,target);
}

const float*  EveFractalMapView::getSystem()     const
{
    return system;
}



void EveFractalMapView::setCamera(const EveCamera<float>&remote)
{
    BEGIN
    float old[3];
    _c3(camera.location,old);
    camera = remote;
    __transformSystem(invert,remote.modelview_invert,camera.modelview_invert);
    __transform(invert,remote.location,camera.location);

    __invertSystem(camera.modelview_invert,camera.modelview);
    
    float speed = _distance(old,camera.location)/eveTiming.delta;
    if (speed > GLOBAL_ERROR)
    {
        _sub(camera.location,old,movement_direction);
        _div(movement_direction,speed*eveTiming.delta);
    }

    velocity = speed;

    if (speed > velocity)
    {
        float fc = 0.5;//pow(1.0f-EVE_MAX_VELOCITY_CHANGE_PER_SECOND,eveTiming.delta);
        velocity = velocity * fc + speed * (1.0f-fc);
    }
    else
        velocity = vmax(speed,velocity*pow(EVE_MAX_VELOCITY_CHANGE_PER_SECOND,eveTiming.delta));
    _c3(camera.location,user_location);
    
    __mult4(camera.projection,camera.modelview,cast_matrix);
    visual_volume = camera.resolveVolume();

    END
}
